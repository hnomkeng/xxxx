function	script	pramool_config	{

"- ปิดประมูล รับเงิน หรือ ไอเทม",closepramool1,
"- ปิดการประมูลแต่เพียงเท่านี้",closepramool2,
"- กำหนดการเปิด - ปิด ประมูล",closepramool3;

additempramool01:
mes "กรุณาใส่ Id ของสินค้า";
input $pramoolitem01;
next;
mes "กรุณาใส่จำนวน ของสินค้า";

next;

input $bitratepramool;
npcshopdelitem "pramoolshop",501;
npcshopadditem "pramoolshop",$pramoolitem01,$startbiterate;

end;

closepramool1:

end;
clpramool:

}
